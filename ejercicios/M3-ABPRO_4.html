CREAR TABLA CAPACITACION

CREATE TABLE CAPACITACION(
    idCapacitacion INT NOT NULL PRIMARY KEY,
    rutCliente VARCHAR(15) NOT NULL,
    dia VARCHAR(25) NULL,
    hora VARCHAR(5) NULL,
    lugar VARCHAR(50) NOT NULL,
    duracion INT NOT NULL,
    cantidadAsistentes INT NULL
);
