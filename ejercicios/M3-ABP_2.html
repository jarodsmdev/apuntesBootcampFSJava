AGREGAR FOREIGN KEY A LA TABLA VENTAS

ALTER TABLE VENTAS
ADD CONSTRAINT ventas_clientes_fk FOREIGN KEY (clientes_idCliente)
REFERENCES CLIENTES (idCliente);

SHOW CREATE TABLE VENTAS;

  `impuesto` int NOT NULL,
